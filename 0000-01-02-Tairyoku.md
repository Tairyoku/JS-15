function maxDiff(arr) {
  if (arr.length <=1) return 0;
 return Math.abs(arr.sort( (a, b) => a - b )[0] 
 - arr.sort( (a, b) => a - b)[arr.length-1]); }

//Проверка
alert(maxDiff([0, 1, 2, 3, 4, 5, 6]));
alert(maxDiff([-0, 1, 2, -3, 4, 5, -6]));
alert(maxDiff([0, 1, 2, 3, 4, 5, 16]));
alert(maxDiff([16]));
alert(maxDiff([]));
